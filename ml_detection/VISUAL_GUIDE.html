Integration Implementation

1. Backend (Python Flask):

from ml_detection.flask_integration import detector_manager

@app.route('/legacy_login', methods=['POST'])
def legacy_login():
  username = request.form.get('username', '')
  
  # Check for attacks
  result = detector_manager.check_input(username)
  
  if result['is_malicious']:
    return jsonify({
      'color': result['color'], # 🔴 RED
      'status': result['status'], # MALICIOUS
      'message': f"Attack detected: {result['attack_type']}"
    }), 403
  
  # Continue with safe input
  return jsonify({'color': '🟢 GREEN', 'status': 'SAFE'}), 200

2. Frontend (JavaScript):

document.getElementById('loginForm').addEventListener('submit', async (e) => {
  e.preventDefault();
  
  const formData = new FormData(e.target);
  const response = await fetch('/legacy_login', {
    method: 'POST',
    body: formData
  });
  
  const data = await response.json();
  
  if (data.status === 'MALICIOUS') {
    // Show RED access denied screen
    showAccessDenied(data.color, data.message);
  } else {
    // Show GREEN success screen
    showSuccess(data.color);
  }
});
